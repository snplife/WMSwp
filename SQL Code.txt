-- Multi-tenant WMS schema + policies (safe migration for existing DB + Android compatibility)

create extension if not exists pgcrypto;

-- 1) Core tables
create table if not exists public.companies (
  id uuid primary key default gen_random_uuid(),
  name text not null unique,
  created_at timestamptz not null default now()
);

create table if not exists public.stock (
  company_id uuid references public.companies (id) on delete cascade,
  position text not null,
  material_code text not null,
  quantity integer not null,
  constraint stock_pk primary key (company_id, position, material_code)
);

create table if not exists public.stock_history (
  event_key text primary key,
  company_id uuid references public.companies (id) on delete cascade,
  action text not null,
  position text not null,
  material_code text not null,
  note text not null default '',
  created_at_ms bigint not null
);

create table if not exists public.app_users (
  user_id uuid primary key references auth.users (id) on delete cascade,
  email text not null,
  username text not null,
  role text not null check (role in ('master', 'user')),
  company_id uuid references public.companies (id),
  created_by uuid references auth.users (id),
  db_url text,
  db_anon_key text,
  updated_at timestamptz not null default now(),
  created_at timestamptz not null default now()
);

-- 2) Add missing columns for existing installations
alter table public.stock add column if not exists company_id uuid references public.companies (id) on delete cascade;
alter table public.stock_history add column if not exists company_id uuid references public.companies (id) on delete cascade;
alter table public.app_users add column if not exists email text;
alter table public.app_users add column if not exists username text;
alter table public.app_users add column if not exists company_id uuid references public.companies (id);
alter table public.app_users add column if not exists created_by uuid references auth.users (id);
alter table public.app_users add column if not exists db_url text;
alter table public.app_users add column if not exists db_anon_key text;
alter table public.app_users add column if not exists updated_at timestamptz not null default now();
alter table public.app_users add column if not exists created_at timestamptz not null default now();

-- 3) Normalize users
update public.app_users
set username = lower(split_part(email, '@', 1))
where (username is null or username = '')
  and email is not null;

-- 4) Default company + backfill company_id for legacy data
insert into public.companies (name)
select 'Default Company'
where not exists (select 1 from public.companies);

update public.stock
set company_id = (select id from public.companies order by created_at asc limit 1)
where company_id is null;

update public.stock_history
set company_id = (select id from public.companies order by created_at asc limit 1)
where company_id is null;

update public.app_users
set company_id = (select id from public.companies order by created_at asc limit 1)
where company_id is null
  and role = 'user';

-- 5) Strengthen constraints after backfill
alter table public.stock alter column company_id set not null;
alter table public.stock_history alter column company_id set not null;
alter table public.app_users
  drop constraint if exists app_users_user_must_have_company;
alter table public.app_users
  add constraint app_users_user_must_have_company
  check (role = 'master' or company_id is not null) not valid;

-- Ensure stock PK includes company_id even on legacy schema
DO $$
DECLARE
  current_pk_name text;
BEGIN
  select conname into current_pk_name
  from pg_constraint
  where conrelid = 'public.stock'::regclass
    and contype = 'p';

  if current_pk_name is not null then
    execute format('alter table public.stock drop constraint %I', current_pk_name);
  end if;

  alter table public.stock
    add constraint stock_pk primary key (company_id, position, material_code);
EXCEPTION
  when duplicate_table then null;
  when duplicate_object then null;
END $$;

-- 7) Helper functions
create or replace function public.touch_app_users_updated_at()
returns trigger
language plpgsql
as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

create or replace function public.fill_company_id_from_auth_user()
returns trigger
language plpgsql
security definer
set search_path = public
as $$
begin
  if new.company_id is null then
    select au.company_id
      into new.company_id
    from public.app_users au
    where au.user_id = auth.uid()
    limit 1;
  end if;
  return new;
end;
$$;

drop trigger if exists trg_touch_app_users_updated_at on public.app_users;
create trigger trg_touch_app_users_updated_at
before update on public.app_users
for each row execute function public.touch_app_users_updated_at();

drop trigger if exists trg_fill_stock_company_id on public.stock;
create trigger trg_fill_stock_company_id
before insert on public.stock
for each row execute function public.fill_company_id_from_auth_user();

drop trigger if exists trg_fill_stock_history_company_id on public.stock_history;
create trigger trg_fill_stock_history_company_id
before insert on public.stock_history
for each row execute function public.fill_company_id_from_auth_user();

create or replace function public.is_master(uid uuid default auth.uid())
returns boolean
language sql
stable
security definer
set search_path = public
as $$
  select exists (
    select 1
    from public.app_users
    where user_id = uid
      and role = 'master'
  );
$$;

create or replace function public.user_company_id(uid uuid default auth.uid())
returns uuid
language sql
stable
security definer
set search_path = public
as $$
  select company_id
  from public.app_users
  where user_id = uid
  limit 1;
$$;

grant execute on function public.is_master(uuid) to authenticated;
grant execute on function public.user_company_id(uuid) to authenticated;

-- 9) Rebuild stock from stock_history events
create or replace function public.apply_stock_history_events(p_company_id uuid default null)
returns void
language plpgsql
security definer
set search_path = public
as $$
begin
  -- Rebuild target scope from history to keep stock consistent.
  if p_company_id is null then
    truncate table public.stock;

    insert into public.stock (company_id, position, material_code, quantity)
    select
      h.company_id,
      h.position,
      h.material_code,
      sum(
        case upper(coalesce(h.action, ''))
          when 'RECEIVE' then 1
          when 'ISSUE' then -1
          when 'MOVE' then 0
          when 'MOVE_ALL' then 0
          when 'ADJUST' then 0
          when 'UNDO' then 0
          else 0
        end
      )::int as quantity
    from public.stock_history h
    where h.company_id is not null
    group by h.company_id, h.position, h.material_code
    having sum(
      case upper(coalesce(h.action, ''))
        when 'RECEIVE' then 1
        when 'ISSUE' then -1
        when 'MOVE' then 0
        when 'MOVE_ALL' then 0
        when 'ADJUST' then 0
        when 'UNDO' then 0
        else 0
      end
    ) > 0;

    return;
  end if;

  delete from public.stock s
  where s.company_id = p_company_id;

  insert into public.stock (company_id, position, material_code, quantity)
  select
    h.company_id,
    h.position,
    h.material_code,
    sum(
      case upper(coalesce(h.action, ''))
        when 'RECEIVE' then 1
        when 'ISSUE' then -1
        when 'MOVE' then 0
        when 'MOVE_ALL' then 0
        when 'ADJUST' then 0
        when 'UNDO' then 0
        else 0
      end
    )::int as quantity
  from public.stock_history h
  where h.company_id = p_company_id
  group by h.company_id, h.position, h.material_code
  having sum(
    case upper(coalesce(h.action, ''))
      when 'RECEIVE' then 1
      when 'ISSUE' then -1
      when 'MOVE' then 0
      when 'MOVE_ALL' then 0
      when 'ADJUST' then 0
      when 'UNDO' then 0
      else 0
    end
  ) > 0;
end;
$$;

grant execute on function public.apply_stock_history_events(uuid) to authenticated;
